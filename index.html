// ELEMENTS const urlInput = document.getElementById("url"); const goBtn = document.getElementById("goBtn"); const iframe = document.getElementById("frame"); // PROXY SETTINGS let proxyEnabled = false; // Choose ONE proxy (none of these are corsproxy.io) let proxyURL = "https://api.codetabs.com/v1/proxy/?quest="; // let proxyURL = "https://thingproxy.freeboard.io/fetch/"; // let proxyURL = "https://api.allorigins.win/raw?url="; // Enable / Disable Proxy function enableProxy() { proxyEnabled = true; console.log("Proxy ON"); } function disableProxy() { proxyEnabled = false; console.log("Proxy OFF"); } // Build proxied URL function tryProxy(url) { return proxyURL + encodeURIComponent(url); } // Load internal Galaxia pages function setTab(page) { iframe.src = page + ".html"; } // Main URL loader function loadURL(value) { let url = value.trim(); // galaxia://page if (url.startsWith("galaxia://")) { const page = url.replace("galaxia://", ""); setTab(page); return; } // Add https:// if missing if (!url.startsWith("http://") && !url.startsWith("https://")) { if (url.includes(".")) { url = "https://" + url; } else { iframe.src = "home.html"; return; } } // Proxy mode iframe.src = proxyEnabled ? tryProxy(url) : url; } // GO button goBtn.addEventListener("click", () => { loadURL(urlInput.value); }); // Press Enter to load urlInput.addEventListener("keydown", (e) => { if (e.key === "Enter") goBtn.click(); }); // Default page setTab("home");
