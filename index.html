let proxyEnabled = false; let proxyURL = "https://corsproxy.io/?"; function toggleProxyPanel() { const panel = document.getElementById("proxyPanel"); panel.style.display = panel.style.display === "none" ? "block" : "none"; } function updateProxyUI() { const box = document.getElementById("proxyPanelStatus"); if (proxyEnabled) { box.textContent = "Proxy: CONNECTED"; box.style.background = "#002200"; box.style.color = "#00ff00"; box.style.border = "1px solid #00ff00"; box.style.boxShadow = "0 0 10px #00ff00"; } else { box.textContent = "Proxy: DISCONNECTED"; box.style.background = "#220000"; box.style.color = "#ff4444"; box.style.border = "1px solid #ff4444"; box.style.boxShadow = "0 0 10px #ff4444"; } } function enableProxy() { proxyEnabled = true; proxyURL = document.getElementById("proxyServer").value; updateProxyUI(); } function disableProxy() { proxyEnabled = false; updateProxyUI(); } // Auto-retry if proxy fails function tryProxy(url) { return proxyURL + encodeURIComponent(url); }
